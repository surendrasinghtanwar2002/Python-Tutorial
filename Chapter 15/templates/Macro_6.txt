            {%- if device.hostname -%}
                hostname {{ device.hostname }} 
            {% endif %}
            router {{ device.routing_protocol }} 
            {%- if device.routing_protocol | lower == "ospf" -%}

                {{ospf_specific_configuration(device)}}     {# Call the OSPF Macro here #}

                router_id {{ device.router_id }}
